class Solution {
    public int[] getConcatenation(int[] nums) {
        int n=nums.length;
        int a[]=new int[n*2];
        for(int i=0;i<n;i++)
        {
            a[i]=nums[i];
        }
        for(int i=n;i<n*2;i++)
        {
            a[i]=nums[i-n];
        }
        return a;
    }
}
