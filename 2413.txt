class Solution {
    public int smallestEvenMultiple(int n) {
        int c=0;
        int f=0;
        for(int i=1;i<=2*n;i++)
        {
            if(i%2==0&&i%n==0)
            {
                   c++;
                   if(c==1)
                   {
                    f=i;
                   }
            }
        }
        return f;
        
    }
}
