class Solution {
    public int pivotInteger(int n) {
    int left = 1;
    int right = n;

    while (left <= right) {
        int mid = (left + right) / 2;

        int sum1 = mid * (mid + 1) / 2; 
        int sum2 = (n * (n + 1) / 2) - sum1 + mid; 

        if (sum1 == sum2) {
            return mid;
        } else if (sum1 < sum2) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }

    return -1;
        
    }
}
