class Solution {
    public int sumOfUnique(int[] nums) {
        int[] k = new int[101]; 
        for (int n : nums) {
            k[n]++;
        }

        int s = 0;
        for (int i = 1; i <= 100; i++) {
            if (k[i] == 1) {
                s += i;
            }
        }

        return s;
    }
}
