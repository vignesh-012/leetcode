class Solution {
    public int mostWordsFound(String[] sentences) {
        int ans=0;
        for(int i=0;i<sentences.length;i++)
        {
            int c=0;
            String s=sentences[i];
            String wc[]=s.split("\\s++");
            c=wc.length;
           ans=Math.max(c,ans);

        }
        return ans;
        
    }
}
