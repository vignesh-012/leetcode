class Solution {
    public int diagonalSum(int[][] mat) {
        int r=mat.length;
        int c=mat[0].length;
        int sum=0;
        for(int i=0;i<r;i++)
        {
            sum=sum+mat[i][i];
        }
        for(int i=0,j=c-1;i<r&&j>=0;i++,j--)
        {
            if(i!=j)
            sum=sum+mat[i][j];
        }
        return sum;
        
    }
}
