class Solution {
    public int differenceOfSum(int[] a) {
        int es=0,sum=0;
        for(int i=0;i<a.length;i++)
        {
            sum=sum+a[i];
            int n=a[i];
            while(n>0)
            {
                int r=n%10;
                es=es+r;
                n=n/10;
            }
            //t=t+es;
            
        }
        return sum-es;
        
    }
}
